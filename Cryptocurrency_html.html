Comparing the price of Solana and Bitcoin over time.

  • Creating a combined graph to show how the prices compare over time together.
In[]:
combined = pd.merge(solana_set, bitcoin_set)
fig, ax1 = plt.subplots(figsize=(13,6))

ax1.plot(combined["Date"], combined["sol"], label = "Solana", color = "Orange")
ax1.set_ylabel("Solana Price", color = "Orange")
ax1.set_xlabel("Date")

ax2 = ax1.twinx()
ax2.plot(combined["Date"], combined["btc"], label = "Bitcoin", color = "Blue")
ax2.set_ylabel("bitcoin Price", color = "Blue")

plt.gca().xaxis.set_major_locator(mdates.WeekdayLocator(interval=15))
plt.title("Comparing Solana and Bitcoin prices")
